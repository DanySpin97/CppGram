147 BasicBot<T>::checkMethodError( const cpr::Response &response, Json::Value &val )
148 {
149  // If there was an error in the connection print it
150  if ( response.error.code != cpr::ErrorCode::OK )
151  {
152  logger_ptr->error( "HTTP Error:" + response.error.message );
153  return false;
154  }
155 
156  if ( !reader.parse( response.text, val ) )
157  {
158  logger_ptr->error( "JSON Parser: Error while parsing JSON document!" );
159  throw JsonParseError();
160  }
161 
162  // Print method error
163  if ( response.status_code != 200 )
164  {
165  logger_ptr->error( "Telegram Error: " + val["error_code"].asString() + ", Description: "
166  + val["description"].asString() );
167  return false;
168  }
169 
170  return true;
171 }

173 template <class T>
174 bool
175 BasicBot<T>::getUpdates( std::vector<Update> &updates,
176  const uint_fast32_t offset,
177  const uint_fast32_t limit,
178  const uint_fast32_t timeout )
179 {
180  // If there are items in the vector
181  if ( updates.size() != 0 )
182  {
183  // clear it
184  updates.clear();
185  }
186 
187  auto response = executeRequest( "getUpdates",
188  cpr::Parameters{{"timeout", to_string( timeout )},
189  {"limit", to_string( limit )},
190  {"offset", to_string( offset + 1 )}} );
191 
192  Json::Value json_updates;
193  if ( !checkMethodError( response, json_updates ) || json_updates["result"].empty() )
194  {
195  return false;
196  }
197 
198  // For each update
199  for ( auto &json_single_update : json_updates["result"] )
200  {
201  // Construct them in place
202  updates.emplace_back( json_single_update );
203  }
204 
205  // We found updates
206  return true;
207 }

449 template <class T>
450 void
451 BasicBot<T>::processUpdate( const Update &update )
452 {
453  if ( command_handler.processCommands( update ) )
454  {
455  return;
456  }
457 
458  switch ( update.type )
459  {
460  case EUpdate::eMessage:
461  {
462  if ( processMessage )
463  {
464  chat_id = to_string( update.message->chat.id );
465  processMessage( dynamic_cast<T &>( *this ), move( update.message.value() ) );
466  }
467  }
468  break;
469  case EUpdate::eCallbackQuery:
470  {
471  chat_id = to_string( update.callback_query->message->chat.id );
472  callback_query_id = update.callback_query->id;
473  processCallbackQuery( *this, move( update.callback_query.value() ) );
474  callback_query_id = "";
475  }
476  break;
477  case EUpdate::eEditedMessage:
478  {
479  if ( processEditedMessage )
480  {
481  chat_id = to_string( update.message->chat.id );
482  processEditedMessage( *this, move( update.message.value() ) );
483  }
484  }
485  break;
486  case EUpdate::eInlineQuery:
487  {
488  chat_id = to_string( update.inline_query->from.id );
489  inline_query_id = update.inline_query->id;
490  processInlineQuery( *this, move( update.inline_query.value() ) );
491  inline_query_id = "";
492  }
493  break;
494  case EUpdate::eChosenInlineResult:
495  {
496  chat_id = to_string( update.chosen_inline_result->from.id );
497  processChosenInlineResult( *this, move( update.chosen_inline_result.value() ) );
498  }
499  break;
500  case EUpdate::eChannelPost:
501  {
502  chat_id = to_string( update.message->chat.id );
503  processChannelPost( *this, move( update.message.value() ) );
504  }
505  break;
506  case EUpdate::eEditedChannelPost:
507  {
508  chat_id = to_string( update.message->chat.id );
509  processEditedChannelPost( *this, move( update.message.value() ) );
510  }
511  break;
512  }
513 }
